package main


type Node struct {
	data interface{}
	next *Node
	prev *Node
}

type List struct {
	len  int
	head *Node
	last *Node
}

// 创建节点
func CreateNode(data interface{}) *Node {
	return &Node{
		data : data,
		next : nil,
		prev : nil,
	}
}

// 创建链表，链表初始化
func CreateList() *List {
	head := CreateNode(nil)
	head.prev = head
	head.next = head
	return &List{
		len  :  0,
		head : head,
		last : nil,
	}
}

// 判空
func (this *List) IsEmpty() bool{
	if this.len == 0 {
		return true
	}
	return false
}
