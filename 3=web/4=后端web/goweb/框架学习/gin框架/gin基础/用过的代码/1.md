```go
package main

import (
	"net/http"

	"github.com/gin-gonic/gin"
)

func SayHello(c *gin.Context) {
	c.JSON(200, gin.H{
		"message": "Hello golang",
	})
}

func main() {

	// 创建一个默认的路由引擎
	r := gin.Default()

	// // get 请求方式： /hello:请求路径
	// // 当客户端以 get 的方法请求 /hello 路径的时候，会执行后面的匿名函数
	// r.GET("/hello",func(c *gin.Context) {
	// 	// c.Json: 返回JSON格式的数据
	// 	c.JSON(200,gin.H {
	// 		"message":"Hello world"
	// 	})
	// })

	//指定
	r.GET("/hello", SayHello)

	r.GET("/book", func(c *gin.Context) {
		c.JSON(200, gin.H{
			"methood": "GET",
		})
	})

	r.POST("/book", func(c *gin.Context) {
		c.JSON(http.StatusOK, gin.H{
			"methood": "POST",
		})
	})

	r.PUT("/book", func(c *gin.Context) {
		c.JSON(http.StatusOK, gin.H{
			"methood": "PUT",
		})
	})

	r.DELETE("/book", func(c *gin.Context) {
		c.JSON(http.StatusOK, gin.H{
			"methood": "DELETE",
		})
	})

	// 启动 HTTP 服务，默许在 0.0.0:8080 启动服务
	r.Run(":9090")
}
```
